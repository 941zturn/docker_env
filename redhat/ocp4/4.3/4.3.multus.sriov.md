```bash

cat << EOF > macvlan.yaml 
apiVersion: operator.openshift.io/v1
kind: Network
metadata:
  name: cluster
spec:
  additionalNetworks: 
  - name: macvlan1
    namespace: demo
    type: SimpleMacvlan
    simpleMacvlanConfig:
      master: eth0
      mode: bridge
      ipamConfig:
        type: static
        staticIPAMConfig:
          addresses:
          - address: 10.1.1.11/24
  - name: macvlan2
    namespace: demo
    type: SimpleMacvlan
    simpleMacvlanConfig:
      master: eth0
      mode: bridge
      ipamConfig:
        type: static
        staticIPAMConfig:
          addresses:
          - address: 10.1.1.12/24
EOF
oc apply -f macvlan.yaml 


oc get networks.operator.openshift.io cluster -o yaml

oc get network-attachment-definitions -n demo

cat << EOF > busybox.yaml 
apiVersion: v1
kind: Pod
metadata:
  name: example-pod
  namespace: demo
  annotations:
    k8s.v1.cni.cncf.io/networks: macvlan1
spec:
  nodeSelector:
    kubernetes.io/hostname: 'worker-0'
  containers:
  - name: example-pod
    command: ["/bin/sh", "-c", "trap : TERM INT; sleep infinity & wait"]
    image: registry.redhat.ren:5443/docker.io/wangzheng422/centos:centos7-test
    imagePullPolicy: Always
EOF
oc apply -f busybox.yaml 

cat << EOF > busybox.yaml 
apiVersion: v1
kind: Pod
metadata:
  name: example-pod2
  namespace: demo
  annotations:
    k8s.v1.cni.cncf.io/networks: macvlan2

  nodeSelector:
    kubernetes.io/hostname: 'infra-0'

  - name: example-pod2

    imagePullPolicy: Always

oc apply -f busybox.yaml 

# https://docs.openshift.com/container-platform/4.3/networking/multiple_networks/configuring-sr-iov.html
oc adm new-project openshift-sriov-network-operator

# create SR-IOV Network Operator from web console

cat << EOF > sriov.yaml
apiVersion: sriovnetwork.openshift.io/v1
kind: SriovNetworkNodePolicy
metadata:
  name: <name> 
  namespace: openshift-sriov-network-operator 
spec:
  resourceName: <sriov_resource_name> 
  nodeSelector:
    feature.node.kubernetes.io/network-sriov.capable: "true" 
  priority: <priority> 
  mtu: <mtu> 
  numVfs: <num> 
  nicSelector: 
    vendor: "<vendor_code>" 
    deviceID: "<device_id>" 
    pfNames: ["<pf_name>", ...] 
    rootDevices: ["<pci_bus_id>", "..."] 
  deviceType: <device_type> 
  isRdma: false 
EOF
```
