# 大内存pod更新image的行为

本文探索如果pod的内存非常大，在更新image定义以后，pod的重启行为表现。

- bilibili
- xigua
- youtube

```bash

cat << EOF > busybox.yaml 
apiVersion: v1
kind: Pod
metadata:
  name: example-pod
spec:
  nodeSelector:
    kubernetes.io/hostname: 'worker-0.ocp4.redhat.ren'
  containers:
  - name: example-pod
    command: ["/bin/sh", "-c", "trap : TERM INT; sleep infinity & wait"]
    image: registry.redhat.ren:5443/docker.io/busybox
    resources:
      requests:
        cpu: 2.0
        memory: 25Gi
      limits:
        cpu: 2.0
        memory: 25Gi
EOF
oc apply -n demo -f busybox.yaml 


cat << EOF > busybox.yaml 
apiVersion: v1
kind: Pod
metadata:
  name: example-pod
spec:
  nodeSelector:
    kubernetes.io/hostname: 'worker-0.ocp4.redhat.ren'
  containers:
  - name: example-pod
    command: ["/bin/sh", "-c", "trap : TERM INT; sleep infinity & wait"]
    image: registry.redhat.ren:5443/docker.io/tomcat
    resources:
      requests:
        cpu: 2.0
        memory: 25Gi
      limits:
        cpu: 2.0
        memory: 25Gi
EOF
oc apply -n demo -f busybox.yaml 


oc delete -n demo -f busybox.yaml 

```
