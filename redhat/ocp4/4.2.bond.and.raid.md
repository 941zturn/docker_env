```yaml
networkd:
  units:
    - name: mvbridge.netdev
      contents: |
        [NetDev]
        Name=mvbridge
        Kind=macvlan

        [MACVLAN]
        Mode=bridge
    - name: vmbridge.network
      contents: |
        [Match]
        Name=vmbridge

        [Network]
        Address=192.168.250.33/24
    - name: ens3.network
      contents: |
        [Match]
        Name=ens3

        [Network]
        MACVLAN=vmbridge
storage:
  disks:
    - device: /dev/sdb
      wipe_table: true
      partitions:
        - label: "raid.1.1"
          number: 0
          size: 0
          start: 0
    - device: /dev/sdc
      wipe_table: true
      partitions:
        - label: "raid.1.2"
          number: 0
          size: 0
          start: 0
  raid:
    - name: data
      level: stripe
      devices:
        - /dev/disk/by-partlabel/raid.1.1
        - /dev/disk/by-partlabel/raid.1.2
  filesystems:
    - name: filesystemdata
      mount:
        device: /dev/md/data
        format: xfs
        wipe_filesystem: true
        label: DATA
systemd:
  units:
    - name: local.volumn.mount
      enable: true
      contents: |
        [Mount]
        What=/dev/md/data
        Where=/data
        Type=xfs

        [Install]
        WantedBy=local-fs.target
```
